/* ================== GLOBAL STATE ================== */
int kcoin = 0;
int maxClickPerDay = 1000;
int clickToday = 0;
bool keyVerified = false;

/* ================== APP ================== */
class MyApp extends StatelessWidget {
  const MyApp({super.key});

  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      debugShowCheckedModeBanner: false,
      home: const MainScreen(),
    );
  }
}

/* ================== MAIN SCREEN ================== */
class MainScreen extends StatefulWidget {
  const MainScreen({super.key});

  @override
  State<MainScreen> createState() => _MainScreenState();
}

class _MainScreenState extends State<MainScreen> {
  int index = 0;

  @override
  void initState() {
    super.initState();
    WidgetsBinding.instance.addPostFrameCallback((_) {
      showKeyDialog();
    });
  }

  void showKeyDialog() {
    TextEditingController keyCtrl = TextEditingController();

    showDialog(
      context: context,
      barrierDismissible: false,
      builder: (_) => AlertDialog(
        title: const Text("Nhập Key"),
        content: Column(
          mainAxisSize: MainAxisSize.min,
          children: [
            TextField(
              controller: keyCtrl,
              decoration: const InputDecoration(hintText: "Nhập key"),
            ),
            const SizedBox(height: 10),
            const Text(
              "Lấy Key free hoặc VIP tại:\n"
              "Zalo: 0854533557\n"
              "Fb: facebook.com/khanhdev223",
              style: TextStyle(fontSize: 12),
            )
          ],
        ),
        actions: [
          TextButton(
            child: const Text("Xác nhận"),
            onPressed: () async {
              bool ok = await verifyKey(keyCtrl.text);
              if (ok) {
                setState(() => keyVerified = true);
                Navigator.pop(context);
              } else {
                ScaffoldMessenger.of(context).showSnackBar(
                  const SnackBar(content: Text("Key không hợp lệ")),
                );
              }
            },
          )
        ],
      ),
    );
  }

  @override
  Widget build(BuildContext context) {
    if (!keyVerified) {
      return const Scaffold(body: SizedBox());
    }

    return Scaffold(
      body: index == 0 ? const HomeScreen() : const ClaimScreen(),
      bottomNavigationBar: BottomNavigationBar(
        currentIndex: index,
        onTap: (i) => setState(() => index = i),
        items: const [
          BottomNavigationBarItem(icon: Icon(Icons.home), label: "Home"),
          BottomNavigationBarItem(icon: Icon(Icons.wallet), label: "Claim Token"),
        ],
      ),
    );
  }
}

/* ================== HOME SCREEN ================== */
class HomeScreen extends StatefulWidget {
  const HomeScreen({super.key});

  @override
  State<HomeScreen> createState() => _HomeScreenState();
}

class _HomeScreenState extends State<HomeScreen>
    with SingleTickerProviderStateMixin {
  late AnimationController controller;

  @override
  void initState() {
    super.initState();
    controller = AnimationController(
      vsync: this,
      duration: const Duration(milliseconds: 500),
    );
  }

  void onClickCoin() {
    if (clickToday >= maxClickPerDay) return;

    setState(() {
      clickToday++;
      kcoin++;
    });

    controller.forward(from: 0);
  }

  @override
  Widget build(BuildContext context) {
    return Column(
      children: [
        header(walletIcon: true),
        const Spacer(),
        ScaleTransition(
          scale: Tween(begin: 1.0, end: 1.2).animate(controller),
          child: GestureDetector(
            onTap: onClickCoin,
            child: Container(
              width: 120,
              height: 120,
              decoration: BoxDecoration(
                shape: BoxShape.circle,
                gradient: const LinearGradient(
                  colors: [Colors.orange, Colors.amber],
                ),
                boxShadow: const [
                  BoxShadow(color: Colors.black26, blurRadius: 10)
                ],
              ),
              child: const Center(
                child: Text(
                  "K",
                  style: TextStyle(
                      fontSize: 48,
                      fontWeight: FontWeight.bold,
                      color: Colors.white),
                ),
              ),
            ),
          ),
        ),
        const SizedBox(height: 20),
        Text("$clickToday / $maxClickPerDay click/day"),
        Padding(
          padding: const EdgeInsets.all(16),
          child: LinearProgressIndicator(
            value: clickToday / maxClickPerDay,
          ),
        ),
        const Spacer(),
      ],
    );
  }
}

/* ================== VERIFY KEY ================== */
Future<bool> verifyKey(String key) async {
  const rawUrl =
      "https://raw.githubusercontent.com/kkkkk5234/toolgopv1/main/key.json";

  final res = await http.get(Uri.parse(rawUrl));
  if (res.statusCode == 200) {
    List list = jsonDecode(res.body);
    return list.contains(key);
  }
  return false;
}
